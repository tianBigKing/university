	function getClickBang(){
		$.getJSON("http://kaoshi.edu.sina.com.cn/interface/abroad_utf8list.php",{'_action' : 'click'},function(data){

				var html = '';

					row = data[i];
					html += '<li><span class="c_count">'+row.click+'</span><span class="num_01">'+(i+1)+'</span><a href="http://kaoshi.edu.sina.com.cn/abroad/'+row.id+'.html" title="'+row.collegename+'" target="_blank">'+row.collegename.substr(0,10)+'</a></li>';
